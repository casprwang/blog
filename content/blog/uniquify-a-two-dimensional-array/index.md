```js
const unique = (arr) => {
	let map = {};
	return arr.filter((item) => {
		if (item in map) return false;
		else {
			map[item] = true;
			return true;
		}
	});
};

	[1, 2],
	[1, 2],
]);
//=> [[1,2]]
```
